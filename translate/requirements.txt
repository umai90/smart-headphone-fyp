# Smart Headphone Translation System
# Install all dependencies:
#   py -m pip install -r requirements.txt

# ── Audio recording & processing ─────────────────────────
# numpy/scipy pinned to match the Pi deployment (see setup_pi.sh) — an
# unpinned librosa/soundfile/numpy/scipy stack produces MEASURABLY different
# feature vectors for the same audio across versions (confirmed: ~30%
# difference in extracted feature magnitude for the same file between
# librosa 0.11.0 and 0.10.2), which changes deepfake-detection verdicts.
# Keep in sync with setup_pi.sh.
sounddevice
numpy==2.4.6
scipy==1.17.1

# ── Offline STT (Vosk for English, Whisper for Urdu) ─────
# Vosk model folder needed:
#   English: vosk-model-small-en-us  (https://alphacephei.com/vosk/models)
# Note: No Vosk model exists for Urdu — Whisper is used instead (auto-downloads on first run).
# Code imports faster_whisper.WhisperModel (mode1_offline_translation.py) — NOT openai-whisper.
vosk
faster-whisper

# ── Offline Translation (Argostranslate) ─────────────────
# Run "Setup Offline Packages" from main menu once (needs internet once).
argostranslate

# ── Offline TTS ──────────────────────────────────────────
pyttsx3

# ── Online STT (Google via SpeechRecognition) ────────────
SpeechRecognition
# webrtcvad enables automatic speech start/stop detection (no fixed 7 s wait).
# On Raspberry Pi OS: sudo apt install gcc python3-dev  — then pip install below.
webrtcvad

# ── Online Translation ────────────────────────────────────
deep-translator

# ── Online TTS ───────────────────────────────────────────
gTTS
pygame-ce

# ── Flask API (for Android app) ──────────────────────────
flask
flask-cors

# ── Deepfake Voice Detection ──────────────────────────────
# Pinned to match the Pi deployment (lcd_ui/setup_pi.sh compiles this exact version
# from source for ARM64). Model .pkl files are only guaranteed to load correctly
# on the scikit-learn version that trained them — keep this pin in sync with
# setup_pi.sh, and re-pin (both here and there) + retrain whenever you upgrade
# scikit-learn in your training environment.
librosa==0.11.0
soundfile==0.13.1
joblib
scikit-learn==1.8.0
matplotlib
# Optional — part of the active ensemble but not fatal if it fails to install
# (no guaranteed aarch64 wheel on all platforms); deepfake_checker.py skips
# individual models that fail to load rather than erroring out.
catboost==1.2.10
